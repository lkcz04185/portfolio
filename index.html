Depth first search

This code snippet example is from the ruby programm that implements depth first search algorithm to find a particular node in the tree using recursion. 


      def dfs(node, searched_node)
        puts node.payload
        return self if node.payload == searched_node
        node.children.each do |child|          
          found = dfs(child, searched_node)
          return found if found
        end
        return nil
      end
      


      def bfs(node, searched_node)
        queue = Queue.new
        queue.enqueue(node)
        until queue == []
          value = queue.dequeue
          puts value.payload      
          if value.payload == searched_node
            return value       
          else
            value.children.each do |child|
              queue.enqueue(child)
            end
          end
        end
      end

Reverse linked list

This is the core method of the ruby programm which reverses the linked list using mutation. 


      def reverse_list(list, previous = nil)
        next_list_node = list.next_node
        list.next_node = previous
        if next_list_node
          reverse_list(next_list_node, list)
        else
          list
        end
      end
